## 工具列表
1. Read - 读取文件内容
2. Write - 创建或覆盖文件
3. Edit - 编辑文件的特定部分
4. MultiEdit - 多处编辑

## 示例代码
const tools = {
    read: "读取文件",
    write: "写入文件",
    edit: "编辑文件",
    multiEdit: "多处编辑"
};

function demonstrateTools() {
    console.log("=== 工具演示开始 ===");
    for (const [name, desc] of Object.entries(tools)) {
        console.log(`  [${name}]: ${desc}`);
    }
    console.log("=== 工具演示结束 ===");
}
