enum class InlineReferenceScheme(val prefix: String, val displayName: String) {
    FILE("file://", "文件"),
    HTTPS("https://", "网页"),
    GIT("git://", "Git仓库")
    // 新类型只需在此添加一行即可

// 自动生成检测模式
fun generateParsePattern(): Regex {
    val prefixes = getAllPrefixes().joinToString("|") { Regex.escape(it) }
    return Regex("@($prefixes[^\\s]+)")
}
